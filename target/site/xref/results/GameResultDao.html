1   package results;

3   import org.jdbi.v3.core.Handle;
4   import org.jdbi.v3.core.Jdbi;
5   import org.jdbi.v3.core.statement.Update;
6   
7   import java.util.List;
8   
9   public class GameResultDao {
10      Handle handle;
11  
12      public GameResultDao(){
13          Jdbi jdbi = Jdbi.create("jdbc:mysql://root:@localhost:3306/snakegame?serverTimezone=Europe/Amsterdam");
14          handle = jdbi.open();
15      }
16  
17      public void insertGameResult(GameResult result){
18          Update insert = handle.createUpdate("INSERT INTO results VALUES (NULL, :playerName, :score, :snakeLength, :numOfPickups, :timeSurvived)");
19          insert.bindBean(result);
20          insert.execute();
21      }
22  
23      public List<GameResult> getGameResults(int limit){
24          return handle.createQuery("SELECT id, playerName, score, snakeLength, numOfPickups, timeSurvived FROM results ORDER BY score DESC LIMIT :limit")
25                  .bind("limit", limit)
26                  .mapToBean(GameResult.class)
27                  .list();
28      }
29  }
